INSERT INTO DISPROD
(PROD_ID, MANUFACTURE_DAT, PROD_NAME, DISPLAY_QUANTITY, 
EXPIRATION_DAT, EVENT_ID)
VALUES (:PROD_ID, :MANUFACTURE_DAT,
	(SELECT PROD_NAME
	FROM PROD
	WHERE PROD_ID = :PROD_ID),
	:QUANTITY, 
	(SELECT EXPIRATION_DAT
	FROM WAREPROD
	WHERE PROD_ID = :PROD_ID AND MANUFACTURE_DAT = :MANUFACTURE_DAT),
	(SELECT EVENT_ID
FROM EPROD
WHERE PROD_ID = :PROD_ID));
