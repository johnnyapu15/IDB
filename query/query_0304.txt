  INSERT INTO WAREPROD
  (QUANTITY, MANUFACTURE_DAT, PROD_ID, PROD_NAME, EXPIRATION_DAT)
  VALUES (:QUANTITY, :MANUFACTURE_DAT, :PROD_ID,
	(SELECT PROD_NAME
	 FROM PROD
	 WHERE PROD_ID = :PROD_ID),
  	 TO_DATE(:MANUFACTURE_DAT) + 
   (SELECT DISTRIB_PERIOD
	 FROM PROD
	 WHERE PROD_ID = :PROD_ID));
