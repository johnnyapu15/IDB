CASE WHEN(:DISPLAY_QUANTITY = 0)
    THEN (DELETE FROM DISPROD
         WHERE (PROD_ID =:PROD_ID) AND (MANUFACTURE_DAT = :MANUFACTURE_DAT))
    ELSE (UPDATE DISPROD
         SET (DISPLAY_QUANTITY = :DISPLAY_QUANTITY)
         WHERE (PROD_ID = :PROD_ID) AND (MANUFACTURE_DAT = :MANUFACTURE_DAT))
END;
