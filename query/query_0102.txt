(ORD_DAT, ORD_ID, PROD_ID, ORD_REQUEST_QUANTITY, DEPOSIT_DAT, ORD_COST, PROD_NAME)

         FROM ORD  
         WHERE (ORD_ID = :ORD_ID))

FROM PROD

FROM PROD
WHERE (PROD_ID = :PROD_ID)));
